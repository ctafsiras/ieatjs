Array Magic

// Mind-blowing array transformation
const numbers = [1, 2, 3, 4, 5];
const magic = numbers
  .map(n => n * 2)
  .filter(n => n > 5)
  .reduce((a, b) => a + b);

console.log(magic); // What's the result? 🤔

Promise Party

async function orderPizza() {
  try {
    await new Promise(r => setTimeout(r, 2000));
    return '🍕 Delivered!';
  } catch {
    return '😢 No pizza today';
  }
}

function createChaos() {
  const box = document.getElementById('chaosBox');
  box.style.transform = 
    `rotate(${Math.random() * 360}deg) 
     scale(${Math.random() + 0.5})`;
}
